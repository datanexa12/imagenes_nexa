actualizar_caso

%let fecha_hora = %sysfunc(datetime());

proc sql;
   update SASDATA.TB_SOLI
   set 
      _Respuesta = "&respuesta",
      Estado_Caso = "&estado",
      Fecha_Fin = &fecha_hora
   where id_caso = &caso_id;
quit;

mostrar_tabla
proc json out=_webout pretty;
  export SASDATA.TB_SOLI;
run;

USUARIO
%global _usuario;
%let _usuario = &SYSUSERID;

/* Crear una tabla con el usuario actual */
data work.usuario_actual;
    length usuario $50;
    usuario = "&_usuario";
run;

/* Exportar como JSON */
proc json out=_webout pretty;
    export work.usuario_actual / nosastags;
run;
